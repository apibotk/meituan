function uploadImage() { if (!previewImage.src) { uploadStatus.textContent = '请先选择或粘贴一张图片'; return; } uploadStatus.textContent = '上传中...'; fetch(previewImage.src) .then(res => res.blob()) .then(blob => { const formData = new FormData(); formData.append('file', blob, 'image.jpg'); fetch('/api/upload', { // 确保这里指向正确的API端点 method: 'POST', body: formData }) .then(response => response.json()) .then(data => { if (data.error) { throw new Error(JSON.stringify(data)); } uploadStatus.textContent = '上传成功！图片链接已自动复制到剪贴板。'; displayResult(data); copyTextToClipboard(data.Jobs); saveToLocalStorage(data); }) .catch(error => { console.error('上传错误:', error); uploadStatus.textContent = '上传失败，请重试。错误详情: ' + error.message; }); }); }
